" Clear all registers
function! ClearRegisters()
    let regs=split('abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ', '\zs')
    for r in regs
    call setreg(r, [])
    endfor
endfunction
nnoremap cr :call ClearRegisters()<enter>
